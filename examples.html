JavaScript Examples

Client-side integration examples using JavaScript.

Basic API Client

class RaghimAIClient {
  constructor(apiUrl, token) {
    this.apiUrl = apiUrl;
    this.token = token;
  }

  async request(endpoint, options = {}) {
    const response = await fetch(`${this.apiUrl}${endpoint}`, {
      headers: {
        'Authorization': `Bearer ${this.token}`,
        'Content-Type': 'application/json',
        ...options.headers
      },
      ...options
    });
    return response.json();
  }

  async uploadDocument(file) {
    const formData = new FormData();
    formData.append('file', file);
    
    const response = await fetch(`${this.apiUrl}/documents/upload`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.token}`
      },
      body: formData
    });
    return response.json();
  }

  async sendQuery(message, collectionId) {
    return this.request('/chat', {
      method: 'POST',
      body: JSON.stringify({
        message,
        collection_id: collectionId
      })
    });
  }
}

Usage Example

// Initialize client
const client = new RaghimAIClient('https://your-domain.com/api', 'your-jwt-token');

// Upload document
const fileInput = document.getElementById('file-input');
const file = fileInput.files[0];
const uploadResult = await client.uploadDocument(file);

// Send query
const response = await client.sendQuery(
  'What is this document about?',
  uploadResult.collectionId
);

console.log(response.message);

Python Examples

Server-side integration examples using Python.

Python Client

import requests
import json

class RaghimAIClient:
    def __init__(self, api_url, token):
        self.api_url = api_url
        self.token = token
        self.headers = {
            'Authorization': f'Bearer {token}',
            'Content-Type': 'application/json'
        }

    def upload_document(self, file_path):
        with open(file_path, 'rb') as f:
            files = {'file': f}
            headers = {'Authorization': f'Bearer {self.token}'}
            response = requests.post(
                f'{self.api_url}/documents/upload',
                files=files,
                headers=headers
            )
        return response.json()

    def send_query(self, message, collection_id):
        data = {
            'message': message,
            'collection_id': collection_id
        }
        response = requests.post(
            f'{self.api_url}/chat',
            json=data,
            headers=self.headers
        )
        return response.json()

    def get_documents(self):
        response = requests.get(
            f'{self.api_url}/documents',
            headers=self.headers
        )
        return response.json()

Usage Example

# Initialize client
client = RaghimAIClient('https://your-domain.com/api', 'your-jwt-token')

# Upload document
result = client.upload_document('document.pdf')
print(f"Document uploaded: {result['documentId']}")

# Send query
response = client.send_query(
    'Summarize this document',
    result['collectionId']
)
print(f"AI Response: {response['message']}")

# List documents
documents = client.get_documents()
for doc in documents:
    print(f"Document: {doc['name']}")

Webhook Examples

Handle real-time events with webhooks.

Webhook Configuration

# Configure webhook
curl -X POST https://your-domain.com/api/webhooks/configure \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "url": "https://your-app.com/webhook",
    "events": ["document.processed", "user.created"],
    "secret": "your-webhook-secret"
  }'

Webhook Handler (Python)

from flask import Flask, request, jsonify
import hmac
import hashlib

app = Flask(__name__)

@app.route('/webhook', methods=['POST'])
def handle_webhook():
    # Verify webhook signature
    signature = request.headers.get('X-Webhook-Signature')
    payload = request.get_data()
    
    expected_signature = hmac.new(
        'your-webhook-secret'.encode(),
        payload,
        hashlib.sha256
    ).hexdigest()
    
    if not hmac.compare_digest(signature, expected_signature):
        return jsonify({'error': 'Invalid signature'}), 401
    
    # Process webhook event
    event = request.json
    event_type = event['type']
    
    if event_type == 'document.processed':
        handle_document_processed(event['data'])
    elif event_type == 'user.created':
        handle_user_created(event['data'])
    
    return jsonify({'status': 'success'})

def handle_document_processed(data):
    print(f"Document processed: {data['documentId']}")

def handle_user_created(data):
    print(f"New user created: {data['userId']}")

if __name__ == '__main__':
    app.run(port=5000)
